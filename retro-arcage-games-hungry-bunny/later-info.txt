# different movements: speed, jump, gravity, flip

# movement parameters 
var speed = 150
var jump_velocity = -300
var gravity = 900

#Persistent velocity
var velocity = Vector2.ZERO # persistent velocity

func _physics_process(delta):
    handle_input()
    apply_gravity(delta)
    move_character()

func handle_input():
    # Horizontal movement
    velocity.x = 0
    if Input.is_action_pressed("ui_right"):
        velocity.x += speed
        $AnimatedSprite2D.flip_h = false
    if Input.is_action_pressed("ui_left"):
        velocity.x -= speed
        $AnimatedSprite2D.flip_h = true

    # Jump
    if is_on_floor() and Input.is_action_just_pressed("ui_up"):
        velocity.y = jump_velocity

    # Animation
    update_animation()

    func apply_gravity(delta):
    velocity.y += gravity * delta

    func move_character():
    velocity = move_and_slide(velocity, Vector2.UP)

    func update_animation():
    if not is_on_floor():
        $AnimatedSprite2D.play("jump")
    elif velocity.x != 0:
        $AnimatedSprite2D.play("walk")
    else:
        $AnimatedSprite2D.play("idle")

# speed-> how fast bunny walks
# jump_velocity-> how high it jumps
# gravity-> pulls the bunny down 
# flip_h-> flips sprite when moving left / right
